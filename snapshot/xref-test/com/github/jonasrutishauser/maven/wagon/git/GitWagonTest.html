51      @DisplayName("openConnectionInternal() opens a GitConnection")
52      void openConnectionInternal_opensGitConnection() {
53          RuntimeException expectedCall = new RuntimeException("expectedCall");
54          GitWagon testee = new GitWagon() {
55  
56              @Override
57              protected GitConnection openGitConnection(GitConfiguration configuration)
58                      throws GitCloneException, GitAuthenticationException {
59                  throw expectedCall;
60              }
61          };
62  
63          RuntimeException exception = assertThrows(RuntimeException.class,
64                  () -> testee.connect(new Repository("foo", "git:test")));
65  
66          assertSame(expectedCall, exception);
67      }

99      @Test
100     @DisplayName("getIfNewer() calls GitConnection.getIfNewer()")
101     void getIfNewer_callsGetIfNewerOnGitConnection() throws Exception {
102         GitConnection gitConnection = mock(GitConnection.class);
103         GitWagon testee = new GitWagon() {
104 
105             @Override
106             protected GitConnection openGitConnection(GitConfiguration configuration)
107                     throws GitCloneException, GitAuthenticationException {
108                 return gitConnection;
109             }
110         };
111         testee.connect(new Repository("foo", "git:test"));
112         File target = new File("target");
113         doReturn(Boolean.TRUE).when(gitConnection).getIfNewer(Paths.get("foo"), Paths.get("target"), 42);
114 
115         assertTrue(testee.getIfNewer("foo", target, 42));
116 
117         verify(gitConnection).getIfNewer(Paths.get("foo"), Paths.get("target"), 42);
118     }
