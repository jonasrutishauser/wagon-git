513     private void addFooFileToRepo( Path tempDir, Optional<Path> pathInRepo, Optional<String> branch )
514         throws IOException, ExecuteException
515     {
516         Path otherWorkingDir = tempDir.resolve( "remote-init" );
517         Files.createDirectory( otherWorkingDir );
518         Executor executor = createExecutor();
519         executor.setWorkingDirectory( otherWorkingDir.toFile() );
520         executor.execute( CommandLine.parse( "git clone ../remote.git ." ) );
521         pathInRepo.map( otherWorkingDir::resolve ).ifPresent( path -> {
522             try
523             {
524                 Files.createDirectories( path );
525             }
526             catch ( IOException e )
527             {
528                 throw new IllegalStateException( e );
529             }
530         } );
531         Files.write( otherWorkingDir.resolve( pathInRepo.orElse( Paths.get( "" ) ) ).resolve( "foo" ),
532                      Arrays.asList( "test" ), StandardCharsets.UTF_8 );
533         executor.execute( CommandLine.parse( "git add ." ) );
534         executor.execute( CommandLine.parse( "git commit -m 'test'" ) );
535         executor.execute( CommandLine.parse( "git push origin master:" + branch.orElse( "master" ) ) );
536     }

594         throws IOException
595     {
596         Path remote = tempDir.resolve( "remote.git" );
597         Files.createDirectory( remote );
598         Executor executor = createExecutor();
599         executor.setWorkingDirectory( remote.toFile() );
600         executor.execute( CommandLine.parse( "git init --bare" ) );
601 
602         Path otherWorkingDir = tempDir.resolve( "remote-init" );
603         Files.createDirectory( otherWorkingDir );
604         executor.setWorkingDirectory( otherWorkingDir.toFile() );
605         executor.execute( CommandLine.parse( "git clone ../remote.git ." ) );
606         executor.execute( CommandLine.parse( "git commit -m 'test' --allow-empty" ) );
607         executor.execute( CommandLine.parse( "git push origin master:" + branch ) );
608 
609         return remote.toUri().toString();
610     }
