44  @DisplayName( "GitConfiguration" )
45  public class GitConfigurationTest
46  {
47  
48      private static class ValidParseArguments
49          implements ArgumentsProvider
50      {
51  
52          private static final List<String> GIT_URLS =
53              Arrays.asList( "file:/repos/test", "ssh://git@github.com/jonasrutishauser/test.git",
54                             "git@github.com:jonasrutishauser/test.git" );
55  
56          private static final List<Optional<String>> BRANCHES =
57              Arrays.asList( Optional.empty(), Optional.of( "master" ), Optional.of( "gh-pages" ),
58                             Optional.of( "feature/test" ) );
59  
60          private static final List<Optional<Path>> PATHS =
61              Arrays.asList( Optional.empty(), Optional.of( Paths.get( "snapshot" ) ),
62                             Optional.of( Paths.get( "snapshot/test" ) ) );
63  
64          @Override
65          public Stream<? extends Arguments> provideArguments( ExtensionContext context )
66          {
67              return GIT_URLS.stream().flatMap( this::addBranch ).flatMap( this::addPaths ).flatMap( this::addInput );
68          }
69  
70          private Stream<Object[]> addBranch( String gitUrl )
71          {
72              return BRANCHES.stream().map( branch -> new Object[] { gitUrl, branch } );
73          }
74  
75          private Stream<Object[]> addPaths( Object[] arguments )
76          {
77              return PATHS.stream().map( path -> new Object[] { arguments[0], arguments[1], path } );
78          }
79  
80          private Stream<? extends Arguments> addInput( Object[] arguments )
81          {
82              Object gitUrl = arguments[0];
83              Optional<?> branch = (Optional<?>) arguments[1];
84              Optional<?> path = (Optional<?>) arguments[2];
85              StringBuilder url = new StringBuilder( "git:" );
86              url.append( gitUrl );
87              if ( branch.isPresent() )
88              {
89                  url.append( '!' ).append( branch.get() );
90              }
91              List<String> urls = new ArrayList<>();
92              if ( path.isPresent() )
93              {
94                  url.append( '!' );
95                  int offset = url.length();
96                  url.append( path.get() );
97                  urls.add( url.toString() );
98                  urls.add( url.toString() + '/' );
99                  url.insert( offset, '/' );
100                 urls.add( url.toString() );
101                 url.append( '/' );
102                 urls.add( url.toString() );
103             }
104             else
105             {
106                 if ( !branch.isPresent() )
107                 {
108                     urls.add( url.toString() );
109                 }
110                 url.append( '!' );
111                 urls.add( url.toString() );
112                 url.append( '/' );
113                 urls.add( url.toString() );
114             }
115             return urls.stream().map( input -> Arguments.of( input, gitUrl, branch, path ) );
116         }
117     }
118 
119     @DisplayName( "parse()" )
120     @ParameterizedTest( name = "{0} => gitUrl={1}, branch={2}, path={3}" )
121     @ArgumentsSource( ValidParseArguments.class )
122     void parse( String url, String gitUrl, Optional<String> branch, Optional<Path> path )
123     {
124         GitConfiguration configuration = GitConfiguration.parse( url );
125 
126         assertAll( () -> assertEquals( gitUrl, configuration.getUrl() ),
127                    () -> assertEquals( branch, configuration.getBranch() ),
128                    () -> assertEquals( path, configuration.getPath() ) );
129     }
130 
131     @DisplayName( "parse() of invalid url" )
132     @ParameterizedTest( name = "{0}" )
133     @ValueSource( strings = { "file:/repos/test", "git:http://foo.bar/test.git!!" } )
134     void parse_invalid( String url )
135     {
136         assertThrows( IllegalArgumentException.class, () -> GitConfiguration.parse( url ) );
137     }
138 
139     @Test
140     @DisplayName( "getWorkingDirectory() returns allways the same directory" )
141     void getWorkingDirectory_returnsAllwaysTheSame()
142         throws IOException
143     {
144         GitConfiguration testee = GitConfiguration.parse( "git:foo" );
145 
146         Path workingDirectory = testee.getWorkingDirectory();
147         try
148         {
149             assertSame( workingDirectory, testee.getWorkingDirectory() );
150         }
151         finally
152         {
153             Files.delete( workingDirectory );
154         }
155     }
